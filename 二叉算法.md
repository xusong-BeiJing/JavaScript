# 普通查找

const index = (val, arr) => {
  for (let i = 0; i < arr.length; i++) {
    if (val === arr[i]) {
      return i;
    }
  }
  return -1;
};
const values = [1, 2, 3, 4, 5, 6, 7];
console.log(index(5, values))


# 二叉算法查找

const index = (val, arr) => {
  let min = 0;
  let max = arr.length - 1;
  while (min <= max) {
    const middle = parseInt((min + max) / 2);
    if (val === arr[middle]) {
      return middle;
    }
    if (val < arr[middle]) {
      max = middle - 1;
    } else {
      min = middle + 1;
    }
  }
  return -1;
};

const values = [1, 3, 4, 7, 9, 10, 11, 14, 17, 20, 23];

console.log(index(14, values));
